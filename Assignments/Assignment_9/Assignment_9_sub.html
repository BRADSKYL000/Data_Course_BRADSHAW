mod1 <- glm(data=df,
            formula = admit ~ gre + gpa + rank,
            family = "binomial")

mod2 <- glm(data=df,
            formula = admit ~ gre + gpa,
            family = "binomial")

mod3 <- glm(data=df,
            formula = admit ~ gre,
            family = "binomial")

mod4 <- glm(data=df,
            formula = admit ~ gpa + rank,
            family = "binomial")

mod5 <- glm(data=df,

mod6 <- glm(data=df,
